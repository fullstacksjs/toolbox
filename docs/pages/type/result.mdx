# result

Represents a success/failure outcome without using exceptions. Pattern commonly used in functional programming.

### Import

```ts copy
import type { Result } from '@fullstacksjs/toolbox';
```

### Signature

```ts copy
type Result<Ok, Err> =
  | { ok: true; value: Ok }
  | { ok: false; error: Err }
```

### Examples

```ts copy
const success: Result<number, string> = { ok: true, value: 42 };
const failure: Result<number, string> = { ok: false, error: 'Something went wrong' };
```

## AsyncResult

A `Promise` wrapper around a `Result` type for async operations. Use for cleaner error handling with `await`.

### Import

```ts copy
import type { AsyncResult } from '@fullstacksjs/toolbox';
```

### Signature

```ts copy
type AsyncResult<Ok, Err> = Promise<Result<Ok, Err>>;
```

### Example

```ts copy
const asyncRes: AsyncResult<number, string> = Promise.resolve({ ok: true, value: 42 });
```

## Ok

Helper function to create a `Result` in the success state.

### Import

```ts copy
import { Ok } from '@fullstacksjs/toolbox';
```

### Signature

```ts copy
function Ok<T>(x: T): Result<T, never>
```

### Example

```ts copy
const success = Ok(42); // { ok: true, value: 42 }
```

## Err

Helper function to create a `Result` in the error state.

### Import

```ts copy
import { Err } from '@fullstacksjs/toolbox';
```

### Signature

```ts copy
function Err<T>(x: T): Result<never, T>
```

### Example

```ts copy
const failure = Err('Something went wrong'); // { ok: false, error: 'Something went wrong' }
```
