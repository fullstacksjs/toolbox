# isPositive

Check whether the given number is positive or not.

### Import

```typescript copy
import { isPositive } from '@fullstacksjs/toolbox';
```

### Signature

```typescript copy
function isPositive(n: number): boolean {};
```

### Examples

```typescript copy
isPositive(0.1)  // true
isPositive(1)    // true
isPositive(10)   // true
isPositive(-1)   // false
isPositive(0)    // false
```
