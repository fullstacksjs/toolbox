# isNegative

Check whether the given number is negative or not.

### Import

```typescript copy
import { isNegative } from '@fullstacksjs/toolbox';
```

### Signature

```typescript copy
function isNegative(n: number): boolean {};
```

### Examples

```typescript copy
isNegative(-0.1)  // true
isNegative(-1)    // true
isNegative(-10)   // true
isNegative(1)     // false
isNegative(0)     // false
```
